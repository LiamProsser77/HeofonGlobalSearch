<h2>🛠️ Browser Integration</h2>

<h3><b>Add Heofon to Chrome:</b></h3>
<ol>
  <li>Open <b>Chrome Settings</b> (<code>chrome://settings/searchEngines</code>).</li>
  <li>Press <b>Manage Search Engines and site search</b>.</li>
  <li>Click <b>Add</b>.</li>
  <li><b>Name:</b> Heofon</li>
  <li><b>Shortcut:</b> heofon</li>
  <li><b>URL:</b> <code>https://heofonglobalsearch-com.base44.appsearch?q=%s</code></li>

        /* LIVE SUGGESTIONS UNDER HOME BOX */
        function updateSuggestions(inputVal) {
            const box = document.getElementById('suggestions');
            const q = inputVal.toLowerCase().trim();

            if (!q) {
                box.innerHTML = '';
                return;
            }

            const matches = [];
            for (let key in searchDB) {
                if (key.startsWith(q)) matches.push(key);
                if (matches.length >= 6) break;
            }

            if (matches.length === 0) {
                box.innerHTML = '';
                return;
            }

            box.innerHTML = matches.map(k => {
                const boldPart = k.slice(0, q.length);
                const restPart = k.slice(q.length);
                return `
                    <div class="suggestion-item" onclick="selectSuggestion('${k.replace(/'/g, "\\'")}')">
                        <div class="suggestion-icon"></div>
                        <div>
                            <span class="suggestion-text-main">${boldPart}</span><span class="suggestion-text-sub">${restPart}</span>
                        </div>
                    </div>
                `;
            }).join('');
        }

        function selectSuggestion(text) {
            const input = document.getElementById('homeSearch');
            input.value = text;
            document.getElementById('suggestions').innerHTML = '';
            performSearch(text);
        }

        /* PERFORM SEARCH */
        function performSearch(query) {
            const q = query.toLowerCase().trim();
            if (!q) return;

            showPage('results-page');
            document.getElementById('resultsSearch').value = query;

            const resultsList = document.getElementById('results-list');
            resultsList.innerHTML = '';

            let found = [];
            for (let key in searchDB) {
                if (key.includes(q)) {
                    found = found.concat(searchDB[key]);
                }
            }

            if (found.length === 0) {
                resultsList.innerHTML = `<p style="color:#bdc1c6;">No results found for "${query}".</p>`;
            } else {
                found.forEach(item => {
                    const div = document.createElement('div');
                    div.className = 'result-item';
                    div.innerHTML = `
                        <div class="result-header">
                            <img src="${item.favicon}" class="result-favicon" onerror="this.style.display='none'">
                            <span>${new URL(item.url).hostname}</span>
                        </div>
                        <a href="${item.url}" class="result-title" target="_blank">${item.name}</a>
                        <p class="result-snippet">${item.desc}</p>
                    `;
                    div.onclick = (e) => {
                        if (e.target.tagName.toLowerCase() !== 'a') {
                            window.open(item.url, '_blank');
                        }
                    };
                    resultsList.appendChild(div);
                });
            }

            const aiTarget = document.getElementById('ai-insight-target');
            aiTarget.innerHTML = '';
            if (insightDB[q]) {
                const card = document.createElement('div');
                card.className = 'ai-insight-card';
                card.innerHTML = `
                    <h3 style="color:#00d4ff; margin-bottom:10px;">Heofon Insight</h3>
                    <p style="color:#e8eaed; line-height:1.6;">${insightDB[q]}</p>
                `;
                aiTarget.appendChild(card);
            } else {
                document.getElementById('side-title').innerText = 'Knowledge Hub';
                document.getElementById('side-content').innerText = 'Search Heofon to unlock deep intelligence insights here.';
            }
        }

        }

        function toggleSafeSearchFromSettings() {
            toggleSafeSearch();
        }

        function setLanguage(lang) {

            document.querySelectorAll('[data-i18n-key]').forEach(el => {
                const key = el.getAttribute('data-i18n-key');
                if (t[key]) el.textContent = t[key];
            });

                const key = el.getAttribute('data-i18n-placeholder');
